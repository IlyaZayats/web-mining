Пример 2: Функции и ошибки (заглушка)
package main

import (
  "errors"
  "fmt"
)

func div(a, b int) (int, error) {
  if b == 0 {
    return 0, errors.New("division by zero")
  }
  return a / b, nil
}

func main() {
  v, err := div(10, 2)
  fmt.Println(v, err)
}
